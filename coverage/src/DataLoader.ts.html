import type { StandardSchemaV1 } from "@standard-schema/spec";
import { okAsync, errAsync, ResultAsync } from "neverthrow";
 
export abstract class DataLoader<TArg> {
  constructor(private readonly data: Record<string, unknown>) {}
 
  private queue: [TArg, (result: boolean) => void][] = [];
  private scheduled = false;
  private cache = new Map<string, Promise<boolean>>();
 
  protected getCacheKey(arg: TArg): string {
    return JSON.stringify(arg);
  }
 
  load(arg: TArg): Promise<boolean> {
    const key = this.getCacheKey(arg);
    if (this.cache.has(key)) {
      return this.cache.get(key)!;
    }
    const promise = new Promise<boolean>((resolve, reject) => {
      this.queue.push([
        arg,
        (result: boolean) => {
          resolve(result);
        },
      ]);
      if (!this.scheduled) {
        this.scheduled = true;
        Promise.resolve().then(() => this.flush());
      }
    });
    this.cache.set(key, promise);
    return promise;
  }
 
  loadMany(args: TArg[]): Promise<boolean[]> {
    return Promise.all(args.map((arg) => this.load(arg)));
  }
 
  private flush(): ResultAsync<void, unknown> {
    // Replace cache for the new batch
    this.cache = new Map();
    const batch = this.queue;
    this.queue = [];
    this.scheduled = false;
    return ResultAsync.fromPromise(
      (async () => {
        await this.processBatch(batch);
      })(),
      (err) => err
    );
  }
 
  abstract processBatch(
    batch: [TArg, resolve: (result: boolean) => void][]
  ): void | Promise<void>;
}
 
export function createDataLoaderFromEvaluateBatch<TArg>(
  evaluateBatchFn: (
    batch: [TArg, resolve: (result: boolean) => void][]
  ) => Promise<void>
) {
  return class extends DataLoader<TArg> {
    async processBatch(batch: [TArg, resolve: (result: boolean) => void][]) {
      await evaluateBatchFn(batch);
    }
  };
}
